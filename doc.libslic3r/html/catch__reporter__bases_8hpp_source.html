167  void sectionStarting( SectionInfo const& sectionInfo ) override {
168  SectionStats incompleteStats( sectionInfo, Counts(), 0, false );
169  std::shared_ptr<SectionNode> node;
170  if( m_sectionStack.empty() ) {
171  if( !m_rootSection )
172  m_rootSection = std::make_shared<SectionNode>( incompleteStats );
173  node = m_rootSection;
174  }
175  else {
176  SectionNode& parentNode = *m_sectionStack.back();
177  auto it =
178  std::find_if( parentNode.childSections.begin(),
179  parentNode.childSections.end(),
180  BySectionInfo( sectionInfo ) );
181  if( it == parentNode.childSections.end() ) {
182  node = std::make_shared<SectionNode>( incompleteStats );
183  parentNode.childSections.push_back( node );
184  }
185  else
186  node = *it;
187  }
188  m_sectionStack.push_back( node );
189  m_deepestSection = std::move(node);
190  }

194  bool assertionEnded(AssertionStats const& assertionStats) override {
195  assert(!m_sectionStack.empty());
196  // AssertionResult holds a pointer to a temporary DecomposedExpression,
197  // which getExpandedExpression() calls to build the expression string.
198  // Our section stack copy of the assertionResult will likely outlive the
199  // temporary, so it must be expanded or discarded now to avoid calling
200  // a destroyed object later.
201  prepareExpandedExpression(const_cast<AssertionResult&>( assertionStats.assertionResult ) );
202  SectionNode& sectionNode = *m_sectionStack.back();
203  sectionNode.assertions.push_back(assertionStats);
204  return true;
205  }
206  void sectionEnded(SectionStats const& sectionStats) override {
207  assert(!m_sectionStack.empty());
208  SectionNode& node = *m_sectionStack.back();
209  node.stats = sectionStats;
210  m_sectionStack.pop_back();
211  }
212  void testCaseEnded(TestCaseStats const& testCaseStats) override {
213  auto node = std::make_shared<TestCaseNode>(testCaseStats);
214  assert(m_sectionStack.size() == 0);
215  node->children.push_back(m_rootSection);
216  m_testCases.push_back(node);
217  m_rootSection.reset();
218 
219  assert(m_deepestSection);
220  m_deepestSection->stdOut = testCaseStats.stdOut;
221  m_deepestSection->stdErr = testCaseStats.stdErr;
222  }
223  void testGroupEnded(TestGroupStats const& testGroupStats) override {
224  auto node = std::make_shared<TestGroupNode>(testGroupStats);
225  node->children.swap(m_testCases);
226  m_testGroups.push_back(node);
227  }
228  void testRunEnded(TestRunStats const& testRunStats) override {
229  auto node = std::make_shared<TestRunNode>(testRunStats);
230  node->children.swap(m_testGroups);
231  m_testRuns.push_back(node);
232  testRunEndedCumulative();
233  }
